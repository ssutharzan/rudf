5 Returning multiple objects

Combine the objects you want into a single object

Example

countA <- function (seq) {
# This function finds the number of 'A's in a given nucleotide sequence
  
  # Convert input sequence to lower case character vector
  seq_lower <- strsplit(as.vector(tolower(seq)),"")[[1]]
  
  # Count the number of As
  a_count <- length(which(seq_lower=="a"))
  
  # Preparing the returning object
  res <- list()
  res$count <- a_count
  res$prop <- a_count/length(seq_lower)

  
  # Returning the A count
  return(res)
}

# Printing the results
nucleotide_seq <- "ATGCaaaTTGG"
count_res <- countA(seq = nucleotide_seq)
cat(paste0("Number of 'A's = ", count_res$count,"\n"))
## Number of 'A's = 4
cat(paste0("Propotion of 'A's = ", count_res$prop,"\n"))
## Propotion of 'A's = 0.363636363636364
